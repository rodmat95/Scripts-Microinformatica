# 1. Definir el dominio y el grupo local de destino
$Dominio = "DOMIBCO"
$GrupoLocal = "Usuarios"
$ComputerName = $env:COMPUTERNAME

# 2. Solicitar el nombre del usuario de red (ID)
$UsuarioRed = Read-Host "Ingrese el ID del usuario de red (ej. 18374001)"

# 3. Solicitar las credenciales del Dominio para consultar Active Directory
Write-Host "`n[REQUERIDO] Ingrese sus credenciales de red para validar el usuario:" -ForegroundColor Cyan
$Credenciales = Get-Credential -UserName "$Dominio\$env:USERNAME" -Message "Buscando datos de $UsuarioRed en el dominio..."

try {
    # 4. Conectarse a Active Directory vía LDAP para extraer el Nombre Completo
    # Extraemos el dominio limpio para estructurar la ruta LDAP (ej: DOMIBCO -> DC=DOMIBCO)
    $RutaLDAP = "LDAP://$Dominio"
    $BuscadorAD = New-Object System.DirectoryServices.DirectorySearcher
    $BuscadorAD.SearchRoot = New-Object System.DirectoryServices.DirectoryEntry($RutaLDAP, $Credenciales.UserName, $Credenciales.GetNetworkCredential().Password)
    $BuscadorAD.Filter = "(&(objectCategory=person)(objectClass=user)(sAMAccountName=$UsuarioRed))"
    
    # Ejecutar búsqueda en la red
    $ResultadoUsuario = $BuscadorAD.FindOne()

    if ($ResultadoUsuario -eq $null) {
        throw "El ID '$UsuarioRed' no existe en el dominio $Dominio."
    }

    # Extraer la propiedad displayName (Nombre Completo)
    $NombreCompleto = $ResultadoUsuario.Properties["displayname"][0]
    
    # 5. Mostrar la validación al administrador y pedir confirmación
    Write-Host "`n--------------------------------------------------" -ForegroundColor Gray
    Write-Host " USUARIO ENCONTRADO EN LA RED" -ForegroundColor Green
    Write-Host " ID de Red:        $UsuarioRed"
    Write-Host " Nombre Completo:  $NombreCompleto" -ForegroundColor White -BackgroundColor DarkBlue
    Write-Host "--------------------------------------------------`n" -ForegroundColor Gray

    # Pregunta Interactiva (OK / Cancelar)
    $Confirmacion = Read-Host "¿Es correcto el usuario? Presione 'O' para Continuar (OK) o cualquier otra tecla para Cancelar"

    if ($Confirmacion.ToUpper() -eq "O") {
        Write-Host "`n[PROCESANDO] Añadiendo usuario al grupo local..." -ForegroundColor Cyan
        
        # 6. Conectarse al grupo local utilizando las mismas credenciales autorizadas
        $ContextoLocal = New-Object System.DirectoryServices.DirectoryEntry("WinNT://$ComputerName/$GrupoLocal,group", $Credenciales.UserName, $Credenciales.GetNetworkCredential().Password)
        $RutaUsuarioRed = "WinNT://$Dominio/$UsuarioRed"
        
        # Ejecutar la inserción
        $ContextoLocal.Invoke("Add", $RutaUsuarioRed)
        
        Write-Host "[ÉXITO] El usuario de red '$NombreCompleto' ($UsuarioRed) se agregó a '$GrupoLocal'." -ForegroundColor Green
    } 
    else {
        Write-Host "`n[CANCELADO] Operación cancelada por el usuario. No se realizaron cambios." -ForegroundColor Yellow
    }
} 
catch {
    Write-Host "`n[ERROR] Proceso fallido o interrumpido." -ForegroundColor Red
    Write-Host "Detalle: $($_.Exception.Message)" -ForegroundColor Yellow
}
