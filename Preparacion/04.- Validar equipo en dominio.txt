# ======================================================
# FASE 2b - PASO 1: Validar que el equipo este en dominio
# ======================================================
# Requisito : Ejecutar como DOMIBCO\AdmLocalSrvWindows
# Proposito : Verificar que el equipo pertenezca al
#             dominio corporativo correcto antes de
#             continuar con los demas pasos.
# ======================================================

$DominioCorrecto = "domibco.com.pe"

# 1. Obtener el dominio actual del equipo
$PropiedadesRed = [System.Net.NetworkInformation.IPGlobalProperties]::GetIPGlobalProperties()
$DominioActual  = $PropiedadesRed.DomainName

# 2. Validar que el equipo pertenezca a algun dominio
if ([string]::IsNullOrWhiteSpace($DominioActual)) {
    Write-Host "[CRITICO] El equipo NO esta unido a un dominio." -ForegroundColor Red
    Write-Host "  -> Esta configurado en un Grupo de Trabajo." -ForegroundColor Yellow
    exit 1
}

# 3. Validar que el dominio sea el correcto
if ($DominioActual.ToLower() -ne $DominioCorrecto.ToLower()) {
    Write-Host "[ALERTA] El equipo esta en un dominio INCORRECTO." -ForegroundColor Red
    Write-Host "  -> Dominio detectado : $DominioActual" -ForegroundColor Yellow
    Write-Host "  -> Dominio esperado  : $DominioCorrecto" -ForegroundColor Cyan
    exit 1
}

Write-Host "[OK] El equipo esta en el dominio correcto: $DominioActual" -ForegroundColor Green
