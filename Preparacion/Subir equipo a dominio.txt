# 1. Solicitar el nuevo nombre de equipo al usuario
$InputHostname = Read-Host "Introduce el nuevo hostname para el equipo"

# 2. Forzar la conversión a mayúsculas
$NuevoHostname = $InputHostname.ToUpper()

# 3. Definir las variables del dominio (hardcodeado) y credenciales
$Dominio = "domibco.com.pe"
$Credenciales = Get-Credential -UserName "$Dominio\Administrador" -Message "Introduce la clave del Administrador del Dominio para unir el equipo $NuevoHostname"

# 4. Cambiar el nombre del equipo y unirlo al dominio
Add-Computer -NewName $NuevoHostname `
             -DomainName $Dominio `
             -Credential $Credenciales `
             -Restart
